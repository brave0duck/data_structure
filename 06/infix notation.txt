/*
문자열을 입력바고 연산자를 만날때까지 하나씩 이동
연산자를 만나면 스택에서 두 요소를 꺼내서 연산을 한다 ( pop())
연산결과를 스택에 푸쉬한다 ( push())

연산자가 없다. or 끝까지 다 돌았다
스택에 있는 결과를 출력한다
*/

typedef enum input_type { LPN, RPN, MLT, DIV, PLS, MNS, OPR} PRECEDENCE;

typedef struct TokenType{
    double value;
    PRECEDENCE type;
}Token;

typedef struct LinkedStackNodeType{
    Token data;
    struct LinkedStackNodeType* pLink;
}
